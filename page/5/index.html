        $nameErr = "请输入姓名";
    } else {
        $name = test_input($_POST['name']);
        if (!preg_match("/^[a-zA-Z]*$/", $name)) {
            $nameErr = "只允许输入字母";
        }
        echo $name;

    if(empty($_POST['mail'])) {

        $website = test_input($_POST['website']);

        $comment = test_input($_POST['comment']);
        echo $comment;

function test_input($data) {
    $data = trim($data);
    $data = stripslashes($data);
    $data = htmlspecialchars($data);
    return $data;
}
